import requests
import json

queue_url = "https://sqs.us-west-2.amazonaws.com/123456789012/MyQueue"

# Define the message body
message_body = json.dumps({"message": "Hello, World!"})

# Make a POST request to the SendMessage endpoint
response = requests.post(queue_url, data={"Action": "SendMessage", "MessageBody": message_body})

# Check the response status code
if response.status_code == 200:
    print("Message sent successfully.")
else:
    print("Failed to send message:", response.text)







curl -X POST https://sqs.<REGION>.amazonaws.com/<ACCOUNT-ID>/<QUEUE-NAME> \
  -H 'Content-Type: application/x-www-form-urlencoded' \
  -H 'Authorization: AWS4-HMAC-SHA256 Credential=<AWS-ACCESS-KEY>/<YYYYMMDD>/<REGION>/sqs/aws4_request, SignedHeaders=content-type;host;x-amz-date, Signature=<SIGNATURE>' \
  -H 'X-Amz-Date: <YYYYMMDDTHHMMSSZ>' \
  -d 'Action=SendMessage&MessageBody=<MESSAGE-BODY>'

Note: Replace the placeholders in the above example with your actual values.

    REGION: the AWS region of your SQS queue
    ACCOUNT-ID: your AWS account ID
    QUEUE-NAME: the name of your SQS queue
    AWS-ACCESS-KEY: your AWS access key ID
    YYYYMMDD: the date in the format YYYYMMDD
    SIGNATURE: the AWS Signature Version 4 signature calculated based on the request
    YYYYMMDDTHHMMSSZ: the current date and time in the format YYYYMMDDTHHMMSSZ
    MESSAGE-BODY: the message body that you want to send to the SQS queue
