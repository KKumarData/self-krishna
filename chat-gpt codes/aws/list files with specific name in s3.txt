#To find a file in an S3 bucket with a specific name, you can use the following Python code:

import boto3

# Replace "bucket_name" with the name of your S3 bucket
bucket_name = "bucket_name"

# Replace "file_name" with the name of the file you are searching for
file_name = "file_name"

# Create an S3 client
s3 = boto3.client('s3')

# List all objects in the bucket
objects = s3.list_objects(Bucket=bucket_name)

# Iterate through the objects and check if the file name matches
for obj in objects['Contents']:
    if obj['Key'] == file_name:
        # The file was found
        print(f"File {file_name} was found in the bucket")

# If the file was not found, print a message
print(f"File {file_name} was not found in the bucket")

#This code will list all objects in the specified S3 bucket and check if any of them have the specified file name. If the file is found, it will print a message indicating that the file was found. If the file is not found, it will print a message indicating that the file was not found.
